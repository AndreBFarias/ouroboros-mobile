## Objetivo

Implementar **9 componentes premium estáticos** que serão usados em
todas as telas. "Estático" aqui significa: não depende de
gesture-handler avançado, não tem física complexa de overlay/sheet,
não tem animação composta entre múltiplos elementos. São os blocos de
base.

Cada componente nasce com:
- Springs no lugar de durations (presets de `@/lib/motion`)
- Haptic light onde aplicável (`@/lib/haptics`)
- Scale 0.97 ao pressionar
- Padding generoso (≥ 20dp lateral em containers)
- Cores vindas de `@/theme/tokens` ou classes Tailwind
- Tipografia mono lowercase em strings de UI

### `src/components/ui/Button.tsx`

```tsx
import { Pressable, Text } from 'react-native';
import { MotiView } from 'moti';
import { useState } from 'react';
import { springs } from '@/lib/motion';
import { haptics } from '@/lib/haptics';

type ButtonVariant = 'primary' | 'success' | 'ghost' | 'destructive';

interface ButtonProps {
  label: string;
  onPress: () => void;
  variant?: ButtonVariant;
  disabled?: boolean;
}

const VARIANT_CLASSES: Record<ButtonVariant, { bg: string; text: string }> = {
  primary: { bg: 'bg-purple', text: 'text-bg' },
  success: { bg: 'bg-green', text: 'text-bg' },
  ghost: { bg: 'bg-transparent border border-bg-elev', text: 'text-fg' },
  destructive: { bg: 'bg-red', text: 'text-fg' },
};

export function Button({
  label,
  onPress,
  variant = 'primary',
  disabled = false,
}: ButtonProps) {
  const [pressed, setPressed] = useState(false);
  const v = VARIANT_CLASSES[variant];

  return (
    <Pressable
      onPressIn={() => {
        if (disabled) return;
        setPressed(true);
        haptics.light();
      }}
      onPressOut={() => setPressed(false)}
      onPress={() => {
        if (!disabled) onPress();
      }}
      disabled={disabled}
      accessibilityRole="button"
      accessibilityLabel={label}
      accessibilityState={{ disabled }}
    >
      <MotiView
        animate={{ scale: pressed ? 0.97 : 1 }}
        transition={springs.snappy}
        className={`${v.bg} rounded-xl py-4 items-center justify-center`}
        style={{ minHeight: 56, opacity: disabled ? 0.4 : 1 }}
      >
        <Text className={`${v.text} font-mono-medium text-base`}>
          {label}
        </Text>
      </MotiView>
    </Pressable>
  );
}
```

### `src/components/ui/PersonAvatar.tsx`

```tsx
import { Pressable, Text } from 'react-native';
import { MotiView } from 'moti';
import { useState } from 'react';
import type { PessoaId } from '@/lib/schemas/pessoa';
import { inicialDe, corDe } from '@/config/pessoas.config';
import { springs } from '@/lib/motion';
import { haptics } from '@/lib/haptics';

type Size = 'sm' | 'md' | 'lg';

interface Props {
  pessoa: PessoaId;
  onPress?: () => void;
  size?: Size;
}

const SIZE_PX: Record<Size, number> = { sm: 28, md: 32, lg: 56 };
const SIZE_TEXT: Record<Size, string> = {
  sm: 'text-xs',
  md: 'text-sm',
  lg: 'text-xl',
};

export function PersonAvatar({ pessoa, onPress, size = 'md' }: Props) {
  const [pressed, setPressed] = useState(false);
  const px = SIZE_PX[size];

  const handle = () => {
    if (onPress) {
      haptics.selection();
      onPress();
    }
  };

  return (
    <Pressable
      onPressIn={() => onPress && setPressed(true)}
      onPressOut={() => setPressed(false)}
      onPress={handle}
      disabled={!onPress}
      accessibilityRole={onPress ? 'button' : 'image'}
      accessibilityLabel={`avatar pessoa ${pessoa}`}
    >
      <MotiView
        animate={{ scale: pressed ? 0.96 : 1 }}
        transition={springs.snappy}
        style={{
          width: px,
          height: px,
          borderRadius: px / 2,
          backgroundColor: corDe(pessoa),
          alignItems: 'center',
          justifyContent: 'center',
        }}
      >
        <Text className={`${SIZE_TEXT[size]} font-mono-medium text-bg`}>
          {inicialDe(pessoa)}
        </Text>
      </MotiView>
    </Pressable>
  );
}
```
